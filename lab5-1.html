class User {
  public $name;
  protected $login;
  private $password;

  public $title, $br;

  public function __construct($name, $login, $password, $title, $br) {
    $this->name = $name;
    $this->login = $login;
    $this->password = $password;
    $this->title = $title;
    $this->br = $br;
  }

  public function __destruct() {
    $this->showInfo($this->title, $this->br);
  }

   public function __clone() {
    $this->name = "Guest";
    $this->login = "guest";
    $this->password = "qwerty";
    $this->title = "Clone User";
    $this->br = " | ";
  }

  public function showInfo($title, $br) {
    $temp = $this->name . $br . $this->login . $br . $this->password;
    echo "<br><b>$title</b><br>"; 
    print ($temp);
  }
}

class SuperUser extends User {
  private $role = "admin";

  public function __construct($name, $login, $password, $title, $br) {
    parent::__construct($name, $login, $password, $title, $br);
  }

  public function showInfo($title, $br) {
    echo "<br><i>$this->role</i>";
    parent::showInfo($title, $br);
  }
}
