# Задание 1-2

  1. Создайте базу данных, отражающую предметную область электронной библиотеки. Необходимо реализовать хранение информации об авторах (ФИО, дата рождения, дата смерти ит.д.), книгах (название, описание, год написания ит.д), жанрах (название жанра и пр.).

  2. Реализуйте связи между таблицами с использованием PRIMARY и FOREIGN KEYS.

# авторы
CREATE TABLE `лаб2`.`авторы` 
( 
	`id` INT NOT NULL AUTO_INCREMENT , 
	`ФИО` VARCHAR(50) NOT NULL , 
	`дата рождения` DATE NOT NULL , 
	`дата смерти` DATE NULL , 

	PRIMARY KEY (`id`)
) 
ENGINE = InnoDB;

# жанры
CREATE TABLE `лаб2`.`жанры` 
( 
	`id` INT NOT NULL AUTO_INCREMENT , 
	`название жанра` VARCHAR(50) NOT NULL , 

	PRIMARY KEY (`id`)
) 
ENGINE = InnoDB;

# книги
CREATE TABLE `лаб2`.`книги` 
( 
	`id` INT NOT NULL AUTO_INCREMENT , 
	`id_автора` INT NULL , 
	`id_жанра` INT NULL , 
	`название` VARCHAR(50) NOT NULL , 
	`описание` TEXT NULL , 
	`год написания` YEAR NOT NULL , 

	PRIMARY KEY (`id`), 
	FOREIGN KEY (`id_автора`) REFERENCES `лаб2`.`авторы`(id) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY (`id_жанра`) REFERENCES `лаб2`.`жанры`(id) ON DELETE CASCADE ON UPDATE CASCADE
) 
ENGINE = InnoDB;

# Задание №4 ❄️ запросы

  1. Выполните следующие запросы:

a. Найти книги, написанные в последнее десятилетие XX века. Вывести название книги, ее жанр и год написания.

SELECT `название`, `id_жанра`, `год написания` FROM `книги`
WHERE `год написания`>2000;


SELECT `название`, `название жанра`, `год написания` FROM `книги`
JOIN `жанры` 
	ON `книги`.`id_жанра` = `жанры`.`id`
WHERE `год написания`>2000;

b. Определить количество книг каждого автора. Вывести имя автора и количество книг.

SELECT `ФИО` as `автор`, COUNT(`id_автора`) as `количество книг` FROM `книги`
JOIN `авторы` 
	ON `книги`.`id_автора` = `авторы`.`id`
GROUP BY `ФИО`;

c. Найти все книги, написанные Агатой Кристи и отсортировать их по годам написания, начиная с самого последнего. Вывести название книги и год написания.

SELECT `название`, `год написания` FROM `книги` 
JOIN `авторы`
	ON `книги`.`id_автора` = `авторы`.`id`
	WHERE `авторы`.`ФИО` LIKE '%Громыко%'
ORDER BY `год написания` DESC;
