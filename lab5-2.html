class User extends AUser {
  public $name;
  protected $login;
  private $password;

  public $objNum;

  public $title, $br;

  static $counter = 0;

    $this->login = $login;
    $this->password = $password;
    $this->title = $title;
    $this->br = $br;
    ++self::$counter;

      if ($name ==  "" || $login == "" || $password == "") throw new Exception("–ü—É—Å—Ç–æ–µ —Å–≤–æ–π—Å—Ç–≤–∞");

        " —Ñ–∞–π–ª–∞ ", $e->getFile();
    }
  }

  public function __toString() {
    return "–û–±—ä–µ–∫—Ç #$this->objNum: $this->name";
  }

  public function __destruct() {
    checkObject($this);    
    $this->showInfo($this->title, $this->br);
  }

  public function showTitle () {
    print (self::INFO_TITLE);
  }

  public function showInfo($title, $br) {
    $this->showTitle();

    $temp = $this->name . $br . $this->login . $br . $this->password;
    echo "<b>$title</b> ($str)<br>";

  }

    return $this->password;
  }

    $this->name = "Guest";
    $this->login = "guest";
    $this->password = "qwerty";

}
